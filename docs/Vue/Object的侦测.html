Jehan 2020-03-20 Vue

# Vue对Object侦测的过程

这里暂时先考虑data中的属性是基本类型的值和object类型的值。

# 1.数据代理

Vue对数据进行数据代理，将data中的属性都绑定到 Vue 的实例上（vm）, 并且给每个属性都绑定了 getter 和 setter，这也就是为什么我们在data中定义的属性可以通过vm实例对象（this）可以访问到，并且还是响应式的。

# 2. 数据劫持

数据劫持的过程就是对data中的所有属性深层次的遍历，都绑定上 getter 和 setter, 使这些属性都成为响应式的。

同时，在这个过程中，我们还要生成一个dep（消息订阅器）来保存页面模板中的依赖（watcher）。核心思想就是在读取属性(触发getter)的时候保存依赖,在修改属性(触发setter)的时候通知依赖去更新界面。

在vue1.x中，页面模板中的一个表达式或者一个指令就是一个watcher（这些表达式和指令的值都是对应的data中的属性）,这样会导致当依赖多的时候，依赖追踪的时候占用太多内存。 从vue2.0开始，加入了虚拟DOM,将颗粒度降为了中等颗粒度，一个组件就是一个watcher，当数据发生变化的时候，通知对应的组件（watcher）,然后在组件内部通过虚拟DOM的diff算法来做最小差异对比，从而更新界面。

function defineReactive (data, key, val) {
  // 每个属性都生成一个dep实例
  const dep = new Dep()
  Object.defineProperty(data, key, {
    // 可枚举
    enumerable: true,
    // 属性描述符可以被修改
    configurable: true,
    get: function () {
      // 添加依赖
      dep.depend()
      return val
    },
    set: function (newVal) {
      if (val === newVal) return 
      val = newVal
      dep.notify()
    }
  })
}

// Dep 类

class Dep () {
  constructor () {
    this.subs = []
  }

  addSub (sub) {
    this.subs.push(sub)
  }

  removeSub (sub) {
    remove(this.subs, sub)
  }

  depend (sub) {
    // 这里target作为一个是否添加watcher的标识，这里假设target就代表一个watcher实例
    if (window.target) {
      this.addSub(window.target)
    }
  }

  notify () {
    // deep copy
    const subs = this.subs.slice()
    for (let i = 0, l = this.subs.length; i < l;i++) {
      // 依次通知watcher的upate方法去更新界面
      subs[i].update()
    }
  }
}

function remove (arr, item) {
    if (arr.length) {
      const index = arr.indexOf(item)
      if (index > -1) {
        return arr.splce(index, 1)
      }
    }
  }

# 3.收集依赖

我们这里收集的依赖其实收集的就是watcher!

当data中的属性发生变化时，触发setter，通知依赖，其实就是通知的watcher,再由watcher去通知其他地方（并不是直接去更新界面，还有一些事情要做，以后再说）。

Watcher类的实现：

class Watcher {
  constructor (vm, expOrFn, cb) {
    // 保存vm实例
    this.vm = vm
    // 得到一个用来获取复杂表达式的值的函数 比如：a.b.c
    this.getter = parsePath(expOrFn)
    // 保存cb
    this.cb = cb
    // 获取值
    this.value = this.get()
  }

  get () {
    // 这里将target赋值为watcher实例，这样当dep.depend()调用的时候，watcher才能被添加进去。
    window.target = this
    // 调用getter函数将vm传入，内部要通过vm来读取表达式的值
    let value = this.getter.call(this,vm, this.vm)
    // 添加完就将target置为null,防止重复添加到dep中去
    window.target = null
    return value
  }

  update () {
    // 缓存旧的值
    const oldValue = this.value
    // 获取最新的值
    this.value = this.get()
    // vm来调用传入watcher的callback,将新值旧值传入
    this.cb.call(this.vm, this.value, oldValue)
  }
}

const bailRE = /[^\w.$]/  // 匹配除了 字母数字下划线汉子并且.结尾的

function parsePath (path) {
  if (bailRE.test(path)) {
    return 
  }
  // 用 . 分割成数组
  const segments = path.split('.')
  return function (obj) {
    for (let i = 0; i < segments.length; i++) {
      if (!obj) return 
      // 循环得到最终的属性值 如：a.b.c
      obj = obj[segments[i]]
    }
  }
}

# 4.递归侦测所有Key

在数据劫持的那里我们提到过，vue初始化的时候要将data的所有属性都要进行深层次的遍历，这个实现对应的就是Observer类。

 class Observer {
   constructor (value) {
     this.value = value
     if (!Array.isArray(value)) {
       this.walk(value)
     }
   }

   walk (obj) {
     const keys = Object.keys(obj)
     for (let i = 0, len = keys.length; i < len; i++) {
       defineReactive(obj, keys[i], obj[keys[i]])
     }
   }

   function defineReactive (data, key, val) {
     // 如果val是object类型，进行深层次的递归遍历，保证所有属性都绑定了getter和setter
     if (typeof val  === 'object') {
       new Observer(val)
     }  
     // 每个属性都生成一个dep实例
    const dep = new Dep()
    Object.defineProperty(data, key, {
      // 可枚举
      enumerable: true,
      // 属性描述符可以被修改
      configurable: true,
      get: function () {
        // 添加依赖
        dep.depend()
        return val
      },
      set: function (newVal) {
        if (val === newVal) return 
        val = newVal
        dep.notify()
      }
    })
  }
 }
