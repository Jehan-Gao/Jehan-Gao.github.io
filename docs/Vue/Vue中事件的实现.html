# $on

/**
 * @param {string | Array<string>} event
 * @param {Function} fn
*/

Vue.prototype.$on = function (event, fn) {
  const vm = this
  // event 如果是数组的情况
  if (Array.isArray(event)) {
    for (let i = 0, l = event.length; i < l; i++) {
      // 依次调用 vm.$on去绑定
      vm.$on(event[i], fn)
    }
  } else {
    // _events[event] 存在，就将fn推入  _events[event] 不存在就创建一个数组，将fn推入
    (vm._events[event] || (vm._events[event] = [])).push(fn)
  }
  return vm
}

# $emit


/**
 * @param {string} event 
*/

Vue.prototype.$emit = function (event) {
  const vm = this
  let cbs = vm._events[event]
  if (cbs) {
    const args = Array.prototype.slice.call(arguments, 1)
    for (let i = 0, l = cbs.length; i < l; i++) {
       cbs[i].apply(vm, args)
    }
  }
  return vm
}

# $off

/**
 * @param {string | Array<string>} event
 * @param {Function} fn
*/

Vue.prototype.$off = function (event, fn) {
  const vm = this
  // 如果没有任何参数 取消所有事件
  if (!arguments.length) {
    vm._events = null
    return vm
  }
  // 如果event 是数组

    return vm

  if (!fn) {
    vm._events[evnet] = null
    return vm
  }

  let cb 
  let i = cbs.length
  while (i--) {
    cb = cbs[i]
    // 特殊处理： cb.fn 专门针对 $once方法的处理
    if (cb === fn || cb.fn === fn) {
      cbs.splice(i, 1)
      break
    }
  }
  return vm
}

# $once

/**
 * @param {string} event
 * @param {Function} fn
*/
Vue.prototype.$once = function (event, fn) {
  const vm = this 
  function on () {
    vm.$off(event, on)
    fn.apply(vm, arguments)
  }
  /*
    特殊处理： 如果使用 $once方法去添加订阅，传入的fn会被替换成on函数
    （on函数对fn函数进行了包装），_evnets[event]保存的是on函数。 
    但是用户并不知道自己传入的fn被替换了，通过 $off(event, fn)去取消fn函数的订阅，此时会删除失败。所以需要将fn挂在on上面,这样就能正确删除了。
  */ 
  on.fn = fn
  vm.$on(event, on)
  return vm
}
