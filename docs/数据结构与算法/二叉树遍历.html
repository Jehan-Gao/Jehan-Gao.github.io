# 二叉树遍历

二叉树遍历也是面试过程中经常遇到的问题。

二叉树遍历的形式有：

  • 前序遍历
  • 中序遍历
  • 后序遍历
  • 层次遍历

其中 前序遍历、中序遍历、后序遍历，这三种遍历方式的区别就是访问根结点的时机不同。

通常前序、中序、后序这三种遍历方式的实现是通过递归来实现的，也可以使用非递归的方式来实现，利用栈-后进先出的特点。

而层次遍历属于广度优先（BFS），利用队列-先进先出的思想。

# 定义一个二叉树结构

function BinaryTree (val) {
  this.val = val
  this.left = this.right = null
}

// 创建结点
// 根结点
var root = new BinaryTree('A')

// 左子树
var leftSubTree = root.left = new BinaryTree('B')
// 右子树
var rightSubTree = root.right = new BinaryTree('C')

// 叶子结点
leftSubTree.left = new BinaryTree('D')
leftSubTree.right = new BinaryTree('E')

rightSubTree.left = new BinaryTree('F')
rightSubTree.right = new BinaryTree('G')

console.log(root)

# 1. 前序遍历

递归实现：

function preorder (root) {
  if (!root) return 
  // 最先访问根结点
  console.log(root.val)
  preorder(root.left)
  preorder(root.right)
}

preorder(root)  // A B D E C F G

非递归实现：

function preorder (root) {
  let stack = []
  stack.push(root)
  while (stack.length) {
    // 从栈顶取出结点
    let top = stack.pop()
    // 输出结点的val
    console.log(top.val)
    // 利用栈后进先出的思想，先压入右子树，后压入左子树
    if (top.right) {
      stack.push(top.right)
    }
    if (top.left) {
      stack.push(top.left)
    }
  }
}
preorder(root) // A B D E C F G

# 2. 中序遍历

递归实现：

function inorder (root) {
  if (!root) return 
  inorder(root.left)
  // 在中间位置访问根结点
  console.log(root.val)
  inorder(root.right)
}

inorder(root) // D B E A F C G

非递归实现：

function inorder (root) {
  let stack = []
  while (true) {
    while (root) {
    // 最开始将根结点压入栈中，然后不断遍历左子树，将左子树压入栈中
      stack.push(root)
      root = root.left
    }
    if (!stack.length) return
    // 将栈顶的结点取出
    let top = stack.pop()
    console.log(top.val)
    // 读取当前结点的右子树，如果没有，则继续从栈顶取元素读取，如果有，则继续进入内部的while中，将右子树压入栈，然后继续遍历左子树，压入栈
    root = top.right
  }
}

inorder(root) // D B E A F C G

# 3. 后序遍历

递归实现：

function postorder (root) {
  if (!root) return 
  postorder(root.left)
  postorder(root.right)
  // 最后访问根结点
  console.log(root.val)
}

postorder(root)  // D E B F G C A

非递归实现：双辅助栈

function postorder (root) {
  let stack = []
  let stack2 = []
  stack.push(root)
  while (stack.length) {
    let top = stack.pop()
    stack2.push(top)
    // 先压入左子树
    if (top.left) {
      stack.push(top.left)
    }
    // 后压入右子树
    if (top.right) {
      stack.push(top.right)
    }
  }
  // 依次遍历输出结点的值
  while (stack2.length) {
    console.log(stack2.pop().val)
  }
}

postorder(root)  // D E B F G C A

# 4. 层次遍历

利用队列的思想（BFS）：

function levelOrder (root) {
  // 准备一个队列
  let queue = []
  // 将根结点推入到队列中
  queue.push(root)
  while (queue.length) {
    // 每次取出第一个结点
    let top = queue.shift()
    // 访问结点
    console.log(top.val)
    if (top.left) {
      queue.push(top.left)
    }
    if (top.right) {
      root = top.right
      queue.push(top.right)
    }
  }
}

levelOrder(root)    // A B C D E F G
